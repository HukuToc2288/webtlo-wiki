---
title: "Расширенные параметры"
weight: 7
---

Есть некоторые параметры config.ini, которые нельзя изменить через графический интерфейс.

```ini
; Настройте формирование вакансий и запустите скрипт cron/vacancies.php
; scan_reports - сканировать списки хранимого (отчёты) по всем подразделам (значения: 0, 1)
; avg_seeders_period - используемый период средних сидов при формировании вакансий (не больше)
; avg_seeders_value - пороговое значение средних сидов на раздаче (не больше)
; reg_time_seconds - пороговое значение время жизни раздачи в секундах (не меньше)
; scan_posted_days - пороговое значение время жизни сообщения с отчётами (не больше)
; exclude_forums_ids - список идентификаторов подразделов через "," исключаемых из вакансий
; include_forums_ids - список идентификаторов подразделов через "," принудительно включаемых в вакансии
; send_topic_id - идентификатор темы на форуме (для отправки вакансий)
; send_post_id - идентификатор сообщения на форуме (для отправки вакансий)
[vacancies]
scan_reports="1" ; по умолчанию: "0" 
scan_posted_days="30" ; по умолчанию: "30"
send_topic_id="4492613" ; по умолчанию: ""
send_post_id="72163293" ; по умолчанию: ""
avg_seeders_period="14" ; по умолчанию: "14"
avg_seeders_value="0.5" ; по умолчанию: "0.5"
reg_time_seconds="2592000" ; по умолчанию: "2592000"
exclude_forums_ids="195,2491,2543" ; по умолчанию: ""
include_forums_ids="" ; по умолчанию: ""

; Настройте отправку отчётов на форум более гибко
; common - выполнять отправку сводного отчёта на форум (значения: 0, 1)
; exclude - список идентификаторов подразделов через "," исключаемых из отправки отчётов на форум
; auto_clear_messages - очищать неактуальные свои сообщения в рабочем подфоруме в старых темах (значения 0, 1)
[reports]
common="0" ; по умолчанию: 1
exclude="123,456,789" ; по умолчанию: ""
auto_clear_messages="1" ; по умолчанию: 0

; Настройте пользовательские таймауты для обращения к форуму и api
; Используйте значение "0" для бесконечного ожидания (не рекомендуется)
; api_timeout - максимально позволенное количество секунд для выполнения cURL-функции при работе с api
; api_connecttimeout - количество секунд ожидания при попытке соединения с api
; forum_timeout - максимально позволенное количество секунд для выполнения cURL-функции при работе с форумом
; forum_connecttimeout - количество секунд ожидания при попытке соединения с форумом
[curl_setopt]
api_timeout="60" ; по умолчанию: 40
api_connecttimeout="60" ; по умолчанию: 40
forum_timeout="60" ; по умолчанию: 40
forum_connecttimeout="60" ; по умолчанию: 40
torrent_client_timeout="60" ; по умолчанию: 40
torrent_client_connecttimeout="60" ; по умолчанию: 40

; Настройте общие параметры
[main]
```
